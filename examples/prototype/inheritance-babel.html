function ParentClass() {
            this.instanceProperty = 'instanceProperty';
            return this;
        }
        ParentClass.staticProperty = 'staticProperty';
        ParentClass.prototype = {
            prototypeProperty: 'prototypeProperty',
        };

        function ChildClass() {
            ParentClass.apply(this, arguments);
            return this;
        }

        function inherits (subClass, superClass) {
            if (typeof superClass !== "function" && superClass !== null) {
                throw new TypeError("Super expression must either be null or a function");
            }

            subClass.prototype = Object.create(superClass && superClass.prototype, {
                constructor: {
                    value: subClass,
                    enumerable: false,
                    writable: true,
                    configurable: true
                }
            });
            if (superClass) {
                if (Object.setPrototypeOf) {
                    Object.setPrototypeOf(subClass, superClass);
                } else {
                    subClass.__proto__ = superClass;
                    console.log('setPrototypeOf not support')
                }
            }
        }

        inherits(ChildClass, ParentClass);

        console.log('ChildClass = ');
        console.log(ChildClass);
        console.log(ChildClass.constructor);
        console.log('ChildClass.staticProperty = ');
        console.log(ChildClass.staticProperty);

        var childClassInstance = new ChildClass();

        console.log('childClassInstance.instanceProperty = ');
        console.log(childClassInstance.instanceProperty);
        console.log('childClassInstance.prototypeProperty = ');
        console.log(childClassInstance.prototypeProperty);
